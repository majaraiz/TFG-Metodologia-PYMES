

!
! ================================================================
! CONFIGURACION SEDE CENTRAL - Agregar cifrado para: Soria
! ================================================================
! Sede REDUNDANTE: Soria tiene 2 routers (principal + backup)
! Aplicar en AMBOS routers de sede central (principal + backup)
!
crypto keyring CLAVE1 vrf main
  pre-shared-key address 172.26.28.51 key C1fr4d0
  pre-shared-key address 172.26.28.52 key C1fr4d0
!
crypto isakmp profile PERFIL1
   match identity address 172.26.28.51 255.255.255.255 main
   match identity address 172.26.28.52 255.255.255.255 main
!
crypto map mapacifrado 2005 ipsec-isakmp
 description Tunel IPSec VRF main contra Soria
 set peer 172.26.28.51
 set peer 172.26.28.52
 set security-association idle-time 600
 set transform-set transformada
 set isakmp-profile PERFIL1
 match address 2005
!
access-list 2005 remark === TUNEL IPSEC VRF main: Sede Central hacia Soria ===
access-list 2005 permit ip 192.168.101.0 0.0.0.255 192.168.5.0 0.0.0.255
!
! ================================================================
! NOTAS PARA SEDE REDUNDANTE:
! ================================================================
! - Soria tiene 2 routers: 172.26.28.51 (principal) + 172.26.28.52 (backup)
! - Ambos routers remotos pueden iniciar tuneles IPSec
! - El failover es automatico via HSRP en la sede remota
! - Esta configuracion se aplica en AMBOS routers de sede central
